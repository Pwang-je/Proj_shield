create table members(

id varchar(20) primary key,

name varchar(20),

passwd varchar(20),

email varchar(40),

phone varchar(30),

area varchar(20),

job varchar(20)) charset=utf8;


create table cases(
no int(5) primary key, 
id varchar(20) not null, 
passwd varchar(20) not null,
area varchar(20),
title varchar(50), 
content text, 
ip varchar(20), 
time varchar(20),
status varchar(20), 
type varchar(20), 
detail varchar(40),
heroes varchar(50)) charset=utf8;
